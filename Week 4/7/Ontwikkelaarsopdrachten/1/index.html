Flexbox

Container Properties

//Creates a flexbox

display: flex

//Direction, establishes main-axis

flex-direction: row || row-reverse || column || column-reverse

//To wrap or not to wrap

flex-wrap: nowrap || wrap || wrap-reverse

//Shorthanded (direction wrap)

flex-flow: flex-direction || flex-wrap

//Alignement main axis

justify-content: flex-start || flex-end || center || space-between || space-around || space-evenly

//Alignement cross axis

align-items: stretch || flex-start || flex-end || center || baseline

//Alignement lines within the container

align-content: normal || flex-start || flex-end || center || space-between || space-around || space-evenly || stretch

Item Properties

//Order of the (flex) html-elements

order: integer (default: 0)

//Ability to grow if necessary

flex-grow: integer (default: 0)

//Ability to chrink if necessary

flex-shrink: positive integer (default: 0)

//Defines default size of the item

flex-basis: length || keyword (default: auto)

//Shorthanded (grow shrink basis)

flex: flex-grow flex-shrink flex-basis (default: 0 1 auto

//Override default alignment from align-items

align-self: auto || felx-start || flex-end || center || baseline || stretch
